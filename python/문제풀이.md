# 3n + 1

```
def gen(n):
	if not n % 2: n /= 2
	else: n = n * 3 + 1
	return int(n)

def get_cycle(num):
	cycleCnt = 0
	while not(num == 1):
		num = gen(num)
		cycleCnt += 1
	return cycleCnt + 1

a, b = input().split()
a, b = int(a), int(b)

max_cycle = 0
for num in range(a, b+1):
	ret = get_cycle(num)
	if ret > max_cycle: max_cycle = ret

print(max_cycle)
```

# 암스트롱 수 구하기
```
# 수 하나를 입력하면 암스트롱 수인지 판별해주는 함수
# 돌려주는 값은 True or False
def isNarcissistic(입력값):
	# 입력값은 숫자 입니다
	입력값 = str(입력값)
	자릿수의합 = 0
	for 자릿수 in 입력값:
		자릿수의합 += int(자릿수) ** 3
	
	if int(입력값) == 자릿수의합: return True
	else: return False

n, m = input().split()
n, m = int(n), int(m)

for i in range(n, m+1):
	if isNarcissistic(i):
		print(i, end=' ')
```
