# 3n + 1

```
def gen(n):
	if not n % 2: n /= 2
	else: n = n * 3 + 1
	return int(n)

def get_cycle(num):
	cycleCnt = 0
	while not(num == 1):
		num = gen(num)
		cycleCnt += 1
	return cycleCnt + 1

a, b = input().split()
a, b = int(a), int(b)

max_cycle = 0
for num in range(a, b+1):
	ret = get_cycle(num)
	if ret > max_cycle: max_cycle = ret

print(max_cycle)
```

# 암스트롱 수 구하기
```
# 수 하나를 입력하면 암스트롱 수인지 판별해주는 함수
# 돌려주는 값은 True or False
def isNarcissistic(입력값):
	# 입력값은 숫자 입니다
	입력값 = str(입력값)
	자릿수의합 = 0
	for 자릿수 in 입력값:
		자릿수의합 += int(자릿수) ** 3
	
	if int(입력값) == 자릿수의합: return True
	else: return False

n, m = input().split()
n, m = int(n), int(m)

for i in range(n, m+1):
	if isNarcissistic(i):
		print(i, end=' ')
```

# stack 문제풀이
```
# 제일 먼저 첫번째 입력으로부터 입력 횟수를 입력 받습니다. 
n = int(input())

stack = []
for _ in range(n):
	cmd = int(input())
	if cmd == 0: # push
		value = int(input()) # push이면 데이터를 추가로 입력
		if len(stack) < 10: # 스택의 크기가 10보다 크면 더 이상 데이터를 입력 X
			stack.append(value)
		else:
			print('overflow')
	elif cmd == 1: # pop
		if not len(stack) == 0:
			stack.pop(-1)
		else:
			print('underflow')
	else: # exit
		break

for data in stack:
	print(data, end= ' ')
```

# 스택 문제풀이(괄호 짝 맞추기)

```
stack = []
flag = True

line = input() # 괄호 짝을 체크할 라인을 읽어 옵니다. 

# 읽어들인 라인으로부터 괄호 문자를 하나씩 얻어온다. 
for braket in line:
	print('current braket: {}'.format(braket))
	# 열려있는 괄호는 전부 스택에 저장(괄호의 종류는 중요하지 않습니다)
	if braket == '(' or braket == '{' or braket == '[':
		stack.append(braket)
		print('push {}'.format(braket))
	else: # 닫힌 괄호들이 들어오는 경우
		if not len(stack) == 0: # 스택이 비어있지 않다면
			print('len: {}, pop: {}'.format(len(stack), stack[-1]))
			if braket == ')': 
				if not (stack.pop(-1) == '('):
					flag = False
					print('flag: {}'.format(flag))
					break	
			elif braket == '}': 
				if not (stack.pop(-1) == '{'):
					flag = False
					print('flag: {}'.format(flag))
					break	
			elif braket == ']': 
				if not (stack.pop(-1) == '['):
					flag = False
					print('flag: {}'.format(flag))
					break	
		else: # 스택이 비어 있다면
			# 스택에 자료가 없는 경우(underflow)는 플래그 변수를 통해 체크 해줍니다.
			print('len: {}, pop: {}'.format(len(stack), stack[-1]))
			flag = False 
			print('flag: {}'.format(flag))
			break

print(flag)
```
