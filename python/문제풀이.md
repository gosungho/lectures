# 3n + 1

```
def gen(n):
	if not n % 2: n /= 2
	else: n = n * 3 + 1
	return int(n)

def get_cycle(num):
	cycleCnt = 0
	while not(num == 1):
		num = gen(num)
		cycleCnt += 1
	return cycleCnt + 1

a, b = input().split()
a, b = int(a), int(b)

max_cycle = 0
for num in range(a, b+1):
	ret = get_cycle(num)
	if ret > max_cycle: max_cycle = ret

print(max_cycle)
```
